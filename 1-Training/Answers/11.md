### Consider the following section of C++ code:

```

#include <iostream>
using namespace std;
int main()
{
    int input;
    cin >> input;
    if (input < 10)
    {
        if (input != 5)
            cout << "wow ";
        else
            input++;
    }
    else
    {
        if (input == 17)
            input += 10;
        else
            cout << "whoa ";
    }
    cout << input << endl;
    return 0;
}
```
